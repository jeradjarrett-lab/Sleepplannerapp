# EyeLoveSleep Production .htaccess
# Copy this file to your server as ".htaccess" (with the dot at the beginning)

# PHP Handler (if needed for your server)
<FilesMatch \.php$>
  SetHandler proxy:fcgi://eyelovesleepcom-php81
</FilesMatch>

# Enable URL Rewriting
<IfModule mod_rewrite.c>
  RewriteEngine On
  RewriteBase /
  
  # Redirect www to non-www (optional)
  # RewriteCond %{HTTP_HOST} ^www\.eyelovesleep\.com [NC]
  # RewriteRule ^(.*)$ https://eyelovesleep.com/$1 [R=301,L]
  
  # Force HTTPS (optional but recommended)
  # RewriteCond %{HTTPS} off
  # RewriteRule ^(.*)$ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]
  
  # Redirect index.html to root
  RewriteCond %{THE_REQUEST} /index\.html [NC]
  RewriteRule ^index\.html$ / [R=301,L]
  
  # Remove trailing slashes
  RewriteCond %{REQUEST_FILENAME} !-d
  RewriteCond %{REQUEST_URI} (.+)/$
  RewriteRule ^ %1 [R=301,L]
  
  # Skip rewrite for existing files
  RewriteCond %{REQUEST_FILENAME} -f
  RewriteRule ^ - [L]
  
  # Skip rewrite for existing directories
  RewriteCond %{REQUEST_FILENAME} -d
  RewriteRule ^ - [L]
  
  # Add .html to clean URLs if file exists
  RewriteCond %{REQUEST_FILENAME}.html -f
  RewriteRule ^(.+)$ $1.html [L]
  
  # Fallback to index.html for everything else
  RewriteRule ^ index.html [L]
</IfModule>

# Browser Caching
<IfModule mod_expires.c>
  ExpiresActive On
  ExpiresByType text/html "access plus 0 seconds"
  ExpiresByType text/css "access plus 1 year"
  ExpiresByType application/javascript "access plus 1 year"
  ExpiresByType image/jpeg "access plus 1 year"
  ExpiresByType image/png "access plus 1 year"
  ExpiresByType image/webp "access plus 1 year"
  ExpiresByType image/svg+xml "access plus 1 year"
  ExpiresByType font/woff2 "access plus 1 year"
</IfModule>

# Cache Control
<IfModule mod_headers.c>
  # No cache for HTML
  <FilesMatch "\.(html|htm)$">
    Header set Cache-Control "no-cache, no-store, must-revalidate"
    Header set Pragma "no-cache"
    Header set Expires "0"
  </FilesMatch>
  
  # Cache static assets
  <FilesMatch "\.(js|css|woff2|jpg|jpeg|png|gif|svg|webp|ico)$">
    Header set Cache-Control "public, max-age=31536000, immutable"
  </FilesMatch>
  
  # Security headers
  Header set X-Content-Type-Options "nosniff"
  Header set X-Frame-Options "SAMEORIGIN"
  Header set X-XSS-Protection "1; mode=block"
</IfModule>

# Compression
<IfModule mod_deflate.c>
  AddOutputFilterByType DEFLATE text/html text/plain text/xml text/css
  AddOutputFilterByType DEFLATE text/javascript application/javascript application/json
  AddOutputFilterByType DEFLATE image/svg+xml
</IfModule>
